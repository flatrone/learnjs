F.prototype

По сути это установка прототипу новые свойства

пример 1

        function Person(name) {
            this.name = name;
          }
          
          // Добавляем новое свойство (метод) к прототипу
          Person.prototype.sayHello = function() {
            console.log(`Hello, my name is ${this.name}`);
          };
          
          let alice = new Person('Alice');
          let bob = new Person('Bob');
          
          // Оба экземпляра наследуют метод sayHello из прототипа
          alice.sayHello(); // "Hello, my name is Alice"
          bob.sayHello();

второй пример

здесь мы говорим - для Rabbit установить все свойства animal`a

        let animal = {
            eats: true
          };
          
          function Rabbit(name) {
            this.name = name;
          }
          
          Rabbit.prototype = animal;
          
          let rabbit = new Rabbit("White Rabbit"); //  rabbit.__proto__ == animal
          
          alert( rabbit.eats ); // true
