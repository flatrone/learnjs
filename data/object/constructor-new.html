Конструктор и оператор New

функция конструктор

function User(name) { this.name = name; this.isAdmin = false; } let user = new User("Jack"); alert(user.name); // Jack alert(user.isAdmin); // false

let obj = new Obj() и let obj = new Obj - тоже самое

создание методов в конструкторе

function User(name) { this.name = name; this.sayHi = function() { alert( "Меня зовут: " + this.name ); }; } let john = new User("John"); john.sayHi(); // Меня зовут: John
